Layer data can be displayed in tables. Tables are defined inside layer.tableview: {} object:

"tableview": {
	"tables": {
	    "Retail Places": {
	    	"from": "dev.retailplaces", // data source
	    	"columns": [
	    	{
	    		"title": "ID",
	    		"field": "id"
	    	},{
	    		"field": "dev_date",
	    		"title": "Date",
	    		"type": "date"
	    	},{
	    		"field": "rp_type"
	    	}
	    	]
	    },
	    ...
	}
}
